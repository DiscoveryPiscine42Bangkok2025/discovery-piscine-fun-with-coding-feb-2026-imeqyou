<!DOCTYPE html>
<html>
<head>
    <title>Balloon</title>

    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
        }

        #balloon {
            width: 200px;
            height: 200px;
            background-color: red;
            border-radius: 50%;
        }
    </style>
</head>

<body>

<div id="balloon"></div>

<script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>

<script>
$(document).ready(function(){

let balloon = $("#balloon");

let size = 200;
let colors = ["red", "green", "blue"];
let index = 0;

balloon.click(function() {
    size += 10;

    if (size > 420) {
        size = 200;
        index = 0;
    } else {
        index = (index + 1) % 3;
    }

    balloon.css({
        width: size + "px",
        height: size + "px",
        backgroundColor: colors[index]
    });
});

balloon.mouseleave(function() {
    size -= 5;

    if (size < 200)
        size = 200;

    index--;
    if (index < 0)
        index = 2;

    balloon.css({
        width: size + "px",
        height: size + "px",
        backgroundColor: colors[index]
    });
});

});
</script>

</body>
</html>
